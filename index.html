<script>

    function checkLogin() {

        const user = document.getElementById('username').value;

        const pass = document.getElementById('password').value;

        // Basic Security: તું અહિયાં તારા પાસવર્ડ સેટ કરી શકે છે

        if(user === "gelexa_team" && pass === "grit2026") {

            document.getElementById('login-box').classList.add('hidden');

            document.getElementById('dashboard').classList.remove('hidden');

        } else {

            alert("Wrong Details! Only Cheetahs allowed.");

        }

    }


    function showForm(type) {

        document.getElementById('dashboard').classList.add('hidden');

        if(type === 'appraisal') document.getElementById('appraisal-form').classList.remove('hidden');

        if(type === 'interview') document.getElementById('interview-form').classList.remove('hidden');

    }


    function goBack() {

        document.getElementById('appraisal-form').classList.add('hidden');

        document.getElementById('interview-form').classList.add('hidden');

        document.getElementById('dashboard').classList.remove('hidden');

    }


    function logout() {

        location.reload();

    }
